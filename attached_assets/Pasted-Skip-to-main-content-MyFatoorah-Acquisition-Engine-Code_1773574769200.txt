// Generate unique merchant hash
function generateMerchantHash(merchant: Merchant): string {
  const normalizedData = {
    name: merchant.name.toLowerCase().trim(),
    website: normalizeUrl(merchant.website),
    phone: normalizePhone(merchant.phone), // +971XXXXXXXXXX
    tradeLicense: merchant.tradeLicense?.toUpperCase(),
    instagram: merchant.instagram?.toLowerCase().replace('@', '')
  };
  
  return sha256(JSON.stringify(normalizedData));
}
